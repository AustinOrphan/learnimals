// Simple Demo Game Implementation
class SimpleDemoGame extends BaseGame {
  constructor(canvasId) {
    super(canvasId, {
      onScoreUpdate: (score) => updateGameStat('score', score),
      onLevelUpdate: (level) => updateGameStat('level', level),
      onStateChange: (state) => updateGameStat('state', state),
      onGameOver: (score) => logEvent(`Game Over! Final Score: ${score}`)
    });
    
    this.circles = [];
    this.spawnTimer = 0;
    this.spawnInterval = 2000; // 2 seconds
  }
  
  onStart() {
    this.circles = [];
    this.spawnTimer = 0;
    logEvent('Game started!');
  }
  
  update(deltaTime, timestamp) {
    // Update spawn timer
    this.spawnTimer += deltaTime;
    if (this.spawnTimer >= this.spawnInterval) {
      this.spawnCircle();
      this.spawnTimer = 0;
    }
    
    // Update circles
    this.circles.forEach((circle, index) => {
      circle.y += circle.speed * deltaTime / 16;
      if (circle.y > this.canvas.height + 50) {
        this.circles.splice(index, 1);
      }
    });
    
    // Update FPS display
    updateGameStat('fps', Math.round(this.getAverageFPS()));
  }
  
  render() {
    super.render(); // Clear canvas
    
    // Draw circles
    this.circles.forEach(circle => {
      this.ctx.fillStyle = circle.color;
      this.ctx.beginPath();
      this.ctx.arc(circle.x, circle.y, circle.radius, 0, Math.PI * 2);
      this.ctx.fill();
    });
  }
  
  handleClick(position) {
    // Check if any circle was clicked
    for (let i = this.circles.length - 1; i >= 0; i--) {
      const circle = this.circles[i];
      const distance = Math.sqrt(
        Math.pow(position.x - circle.x, 2) + 
        Math.pow(position.y - circle.y, 2)
      );
      
      if (distance <= circle.radius) {
        this.circles.splice(i, 1);
        this.addScore(10);
        this.playSound(600, 100);
        break;
      }
    }
  }
}
